Call super method with arguments as array

Thrown when method is not implemented in super class

Returns:

Type

Example

$class('mygame.SpecialKey', 'mygame.Key', {
   __construct: function(x, y, highlight) {
      this.applySuper(mygame.Key, '__construct', [ x, y ]);
      this.highlight = highlight || 'red';
   },
   render: function(dc) {
      this.applySuper(mygame.Key, 'render', [ dc ]);

      // Add highlight square at top-left of key graphic
      dc.fillStyle = this.highlight;
      dc.fillRect(this.x, this.y, 2, 2);
   }
});
