Example

Attach widget to DOM element

jQuery(function() {
   // Initialise widget
   jQuery('#my-widget').widget('init', MyWidget);

   // Initialise widget with custom options
   jQuery('#my-widget').widget('init', MyWidget, {
      "color": "green"
   });
});

Ex#1. Invoke widget method

jQuery('#my-widget').widget('detach');

Ex#3. Getter and setter methods

// Invoke "color" getter method

// Invoke "color" setter method
color = jQuery('#my-widget').widget('color'); // green
